Group network

Based on shared members

group <- table(cesab$group, cesab$name)
member_overlap <- as.matrix(
  proxy::simil(group, method = "russel", diag = TRUE, upper = TRUE) *
    ncol(group)
)
diag(member_overlap) <- 0

netMEM <- igraph::graph_from_adjacency_matrix(
  member_overlap,
  mode = "undirected",
  weighted = TRUE
)
# plot(

Member network

Based on group membership

pw <- c()

for (i in sort(unique(cesab$group))) {
  ni <- cesab$name[cesab$group == i]
  pwi <- expand.grid(ni, ni)
  #remove identity / loop
  pwi <- pwi[pwi$Var1 != pwi$Var2, ]
  pw <- rbind(pw, pwi)
}

netRES <- igraph::graph_from_data_frame(pw, directed = FALSE)

# plot(
#   netRES,
#   layout = layout_with_mds(netRES),
#   vertex.label = NA
# )

vcount(netRES) # 885 nodes : members
